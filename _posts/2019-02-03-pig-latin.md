```py
import pytest


def rindex_alnum(mystr):
    '''Return index of the last alphanumeric character.'''
    slen = len(mystr) - 1
    while slen > 0:
        if not mystr[slen].isalnum():
            slen -= 1
        else:
            break
    return slen


def solution(mystr):
    if not mystr:
        return mystr
    last_alnum = rindex_alnum(mystr)
    trimmed = mystr[:last_alnum + 1]
    end = mystr[last_alnum + 1:] if trimmed != mystr else ''
    trimmed = ' '.join(f'{x[1:]}{x[0]}ay' for x in trimmed.split(' '))
    return f'{trimmed}{end}'


def test_mytest():
    assert solution('') == ''
    assert solution('a') == 'aay'
    assert solution('ab') == 'baay'
    assert solution('foo') == 'oofay'
    assert solution('pig') == 'igpay'
    assert solution('foo bar') == 'oofay arbay'
    assert solution('foo bar !') == 'oofay arbay !'


def test_provided():
    assert solution('Hello world !') == 'elloHay orldway !'
    assert solution('Pig latin is cool') == 'igPay atinlay siay oolcay'


pytest.main([__file__])
```
